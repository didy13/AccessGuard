Set-ExecutionPolicy -ExecutionPolicy RemoteSigned -Scope Process -Force
Import-Module Microsoft.Graph.Authentication
Import-Module Microsoft.Graph.Applications
Import-Module Microsoft.Graph.Users
Import-Module Microsoft.Graph.Identity.SignIns
$tenantId = "09dd8e48-9141-432a-b9c5-b5dd6b9073e6"
$clientAppId = "85884287-2d12-488e-a24a-1786ce76a689"
$resourceAppId = "00000003-0000-0000-c000-000000000000"
$permissions = @("User.Read", "Mail.Read", "offline_access")
$userEmails = @(
    "nikolapetrovic140999@st3fkeeegmail.onmicrosoft.com",
    "nemanjazivkovic060181@st3fkeeegmail.onmicrosoft.com",
    "marko.markovic110200@st3fkeeegmail.onmicrosoft.com",
    "katarinagavrilovic030887@st3fkeeegmail.onmicrosoft.com",
    "vanjaristic130900@st3fkeeegmail.onmicrosoft.com",
    "bojanakrstic090794@st3fkeeegmail.onmicrosoft.com"
)
Write-Host "Povezivanje na Microsoft Graph..." -ForegroundColor Cyan
Connect-MgGraph -TenantId $tenantId -Scopes "User.ReadBasic.All", "Application.ReadWrite.All", "DelegatedPermissionGrant.ReadWrite.All", "AppRoleAssignment.ReadWrite.All"
Write-Host "Dohvatanje Service Principal-a..." -ForegroundColor Cyan
$clientSp = Get-MgServicePrincipal -Filter "appId eq '$clientAppId'"
if (-not $clientSp) {
    Write-Host "Kreiram Service Principal za klijentsku aplikaciju..." -ForegroundColor Yellow
    $clientSp = New-MgServicePrincipal -AppId $clientAppId
}
$resourceSp = Get-MgServicePrincipal -Filter "appId eq '$resourceAppId'"
$scopeToGrant = $permissions -join " "
foreach ($email in $userEmails) {
    Write-Host "Procesuiram: $email" -ForegroundColor Cyan

    $user = Get-MgUser -UserId $email -ErrorAction SilentlyContinue
    if (-not $user) {
        Write-Host "❌ Korisnik $email nije pronađen u Azure AD. Preskačem." -ForegroundColor Red
        continue
    }


    $existingGrant = Get-MgOauth2PermissionGrant -Filter "clientId eq '$($clientSp.Id)' and principalId eq '$($user.Id)' and consentType eq 'Principal'"

    if ($existingGrant) {
        Write-Host "ℹ️ Grant već postoji za $email. Proveri da li treba ažurirati." -ForegroundColor Yellow
        # Ovde možeš dodati logiku za ažuriranje ako želiš, ali za sada preskačemo.
        continue
    }

    try {
        $grant = New-MgOauth2PermissionGrant -ResourceId $resourceSp.Id `
                                             -Scope $scopeToGrant `
                                             -ClientId $clientSp.Id `
                                             -ConsentType "Principal" `
                                             -PrincipalId $user.Id
        Write-Host "✅ Grant kreiran za $email" -ForegroundColor Green
    } catch {
        Write-Host "❌ Greška pri kreiranju granta za $email : $($_.Exception.Message)" -ForegroundColor Red
    }
}
